#include <iostream>

class Base {
public:
    void show() {
        std::cout << "Base class" << std::endl;
    }
};

class Derived1 : virtual public Base {
    // ...
};

class Derived2 : virtual public Base {
    // ...
};

class DerivedFinal : public Derived1, public Derived2 {
    // ...
};

int main() {
    DerivedFinal obj;
    obj.show(); // Ora funziona correttamente, non c'e' ambiguita'
    return 0;
}
